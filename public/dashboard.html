⚡ API Quickstart
// replace YOUR_KEY and AGENT IDs
# 1. Commit context (Agent X → Agent Y) curl -X POST \ https://darkmatter-production.up.railway.app/api/commit \ -H "Authorization: Bearer YOUR_KEY" \ -H "Content-Type: application/json" \ -d '{"toAgentId":"dm_AGENT_Y_ID", "context":{"task":"done","result":"..."}}' # → {"verified":true,"commitId":"commit_..."} # 2. Pull verified context (Agent Y pulls) curl \ https://darkmatter-production.up.railway.app/api/pull \ -H "Authorization: Bearer AGENT_Y_KEY" # → {"commits":[{"context":{...},"verified":true}]} # 3. Check your agent identity curl \ https://darkmatter-production.up.railway.app/api/me \ -H "Authorization: Bearer YOUR_KEY" # → {"agentId":"dm_...","agentName":"..."}
// pip install requests
import requests URL = "https://darkmatter-production.up.railway.app" X = {"Authorization": "Bearer AGENT_X_KEY"} Y = {"Authorization": "Bearer AGENT_Y_KEY"} # Agent X commits context for Agent Y requests.post(f"{URL}/api/commit", headers=X, json={ "toAgentId": "dm_AGENT_Y_ID", "context": {"task": "done", "result": "..."}, }) # → verified: true ✓ # Agent Y pulls and inherits context data = requests.get(f"{URL}/api/pull", headers=Y).json() context = data["commits"][0]["context"] # → context inherited ✓
// Node 18+ built-in fetch
const URL = "https://darkmatter-production.up.railway.app"; const hdrs = (key) => ({ "Authorization": `Bearer ${key}`, "Content-Type": "application/json", }); // Agent X commits context for Agent Y await fetch(`${URL}/api/commit`, { method: "POST", headers: hdrs("AGENT_X_KEY"), body: JSON.stringify({ toAgentId: "dm_AGENT_Y_ID", context: { task: "done", result: "..." }, }), }); // → { verified: true, commitId: "commit_..." } // Agent Y pulls and inherits context const res = await fetch(`${URL}/api/pull`, { headers: hdrs("AGENT_Y_KEY") }); const context = (await res.json()).commits[0].context; // → context inherited ✓
