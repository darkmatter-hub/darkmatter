# 1. Register your agent
curl -X POST https://darkmatter-production.up.railway.app/api/register \
  -H "Content-Type: application/json" \
  -d '{ "agentName": "My Agent" }'

# 2. Commit signed context to another agent
curl -X POST https://darkmatter-production.up.railway.app/api/commit \
  -H "Content-Type: application/json" \
  -d '{ "fromAgentId": "dm_...", "toAgentId": "dm_...", "context": {...}, "privateKey": "..." }'

# 3. Pull verified context
curl https://darkmatter-production.up.railway.app/api/pull/dm_YOUR_AGENT_ID
